# ==========================================
# APPLICATION CONFIGURATION
# ==========================================
NODE_ENV=production
PORT=3001
APP_NAME=your-service-name

# ==========================================
# DOMAIN CONFIGURATION
# ==========================================
# Replace with your actual domain
DOMAIN=yourdomain.com
API_DOMAIN=api.yourdomain.com
COOKIE_DOMAIN=yourdomain.com

# Allowed origins (comma-separated)
ALLOWED_ORIGINS=https://yourdomain.com,https://api.yourdomain.com

# ==========================================
# DATABASE CONFIGURATION
# ==========================================

# MongoDB Connection
MONGODB_URI=mongodb://your_db_user:YOUR_DB_PASSWORD@localhost:27017/your_database?authSource=your_database
MONGODB_DATABASE=your_database

# MongoDB Admin (for backups only, not for apps)
MONGO_ROOT_PASSWORD=YOUR_SECURE_MONGODB_ROOT_PASSWORD

# ==========================================
# MESSAGE QUEUE (RabbitMQ)
# ==========================================
RABBIT_USERNAME=admin
RABBIT_PASSWORD=YOUR_SECURE_RABBITMQ_PASSWORD
RABBIT_URL=localhost:5672
RABBIT_QUEUE_NAME=your-service-queue

# ==========================================
# AUTHENTICATION & SECURITY
# ==========================================

# JWT Configuration (generate with: openssl rand -base64 48)
JWT_SECRET=YOUR_SECURE_JWT_SECRET_HERE
JWT_EXPIRES_IN=3600          # 1 hour in seconds
JWT_REFRESH_EXPIRES_IN=604800 # 7 days in seconds

# Cookie Configuration (in milliseconds)
COOKIE_TOKEN_EXPIRE=3600000          # 1 hour
COOKIE_REFRESH_TOKEN_EXPIRE=604800000 # 7 days

# ==========================================
# EMAIL CONFIGURATION (Optional)
# ==========================================
SMTP_HOST=smtp.youremailprovider.com
SMTP_PORT=587
SMTP_USER=your_email@example.com
SMTP_PASSWORD=YOUR_EMAIL_PASSWORD
EMAIL_FROM=noreply@yourdomain.com

# ==========================================
# LOGGING & MONITORING
# ==========================================
LOG_LEVEL=info
LOG_HTTP_REQUESTS=true

# ==========================================
# RATE LIMITING
# ==========================================
RATE_LIMIT_WINDOW=900000  # 15 minutes
